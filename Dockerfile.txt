# ---------- Build Stage ----------
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copy the project file and restore dependencies
COPY ["USERLOGIN_DEMO/USERLOGIN_DEMO.csproj", "USERLOGIN_DEMO/"]
RUN dotnet restore "USERLOGIN_DEMO/USERLOGIN_DEMO.csproj"

# Copy all files and build the project
COPY . .
WORKDIR "/src/USERLOGIN_DEMO"
RUN dotnet publish "USERLOGIN_DEMO.csproj" -c Release -o /app/publish /p:UseAppHost=false

# ---------- Runtime Stage ----------
FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS runtime
WORKDIR /app

# Copy published files from build stage
COPY --from=build /app/publish ./

# Bind to Render dynamic port
ENV ASPNETCORE_HTTP_PORTS=${PORT}

# Run the application
ENTRYPOINT ["dotnet", "USERLOGIN_DEMO.dll"]
